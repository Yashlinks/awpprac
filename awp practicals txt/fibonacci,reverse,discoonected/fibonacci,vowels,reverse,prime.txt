using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.UI;
using System.Web.UI.WebControls;

namespace _1C__fibonnaci_prime_vowels_looparray_reverse
{
    public partial class WebForm1 : System.Web.UI.Page
    {
        double no1, no2, no3, no;
        String op = "";

        protected void Page_Load(object sender, EventArgs e)
        {

        }
        protected void fibonacci_btn_Click(object sender, EventArgs e)
        {
            no = Convert.ToInt32(fibonacci_txt.Text);
            no1 = 0;
            no2 = 1;
            op = "Fibonacci series-" + no1 + "," + no2 + ",";
            for (int i = 2; i <= no; i++)
            {
                no3 = no1 + no2;
                op += no3 + ",";
                no1 = no2;
                no2 = no3;
            }
            fibonacciresult_lbl.Text = op;
        }
        protected void primechk_btn_Click(object sender, EventArgs e)
        {
            int n = int.Parse(prime_txt.Text);
            int i = 2;
            while(i<=n-1)
            {
                if(n%i==0)
                {
                    primeresult_lbl.Text = n + "Is not a Prime No";
                    break;
                }
                i++;
            }
            if (i == n)
                primeresult_lbl.Text = n + "Is a Prime No";

        }

        protected void Vowels_btn_Click(object sender, EventArgs e)
        {
            string str;
            int i, len, vowel;
            str = vowels_txt.Text;
            vowel = 0;
            len = str.Length;
            for (i = 0; i < len; i++)
            {
                if (str[i] == 'a' || str[i] == 'e' || str[i] == 'i' || str[i] == 'o' || str[i] == 'u' || str[i] == 'A' || str[i] == 'E' || str[i] == 'I' || str[i] == 'O' || str[i] == 'U')
                {
                    vowel++;
                    vowelsresult_lbl.Text = "The No of Vowels in the string is" + vowel;
                }
                else if ((str[i] >= 'a' && str[i] <= 'z') || (str[i] >= 'A' && str[i] <= 'Z')) ;
            }
        }


       

        protected void reversesum_btn_Click(object sender, EventArgs e)
        {
            int n, m, r = 0, d;
            n=int.Parse(reverse_txt.Text);
            m = n;
            while(n>0)
            {
                d = n % 10;
                r = r * 10 + d;
                n = n / 10;
            }
            reverse_lbl.Text = "Reverse of" + m + "=" + r;
        }

    }
}
